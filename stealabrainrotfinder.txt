-- 📍 НАЙТИ СВОЙ ПЛОТ
local closestPlot = nil
local closestDistance = math.huge

for _, plot in ipairs(plotsFolder:GetChildren()) do
    local part = plot:FindFirstChildWhichIsA("BasePart")
    if part then
        local distance = (part.Position - hrp.Position).Magnitude
        if distance < closestDistance then
            closestDistance = distance
            closestPlot = plot
        end
    end
end

if closestPlot then
    print("🚫 ТВОЙ ПЛОТ:", closestPlot.Name)
end

-- ✨ ESP
local function addESP(obj, color)
    if obj:FindFirstChild("ESP") then return end

    -- 🌟 Highlight
    local h = Instance.new("Highlight")
    h.Name = "ESP"
    h.FillColor = color
    h.OutlineColor = color
    h.FillTransparency = 0.5
    h.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    h.Parent = obj

    -- 🏷️ Billboard (имя)
    local billboard = Instance.new("BillboardGui")
    billboard.Name = "ESP_Name"
    billboard.Size = UDim2.new(0, 200, 0, 50)
    billboard.AlwaysOnTop = true
    billboard.StudsOffset = Vector3.new(0, 3, 0)

    -- пытаемся прикрепить к основной части
    local adornee = obj:IsA("Model") and obj:FindFirstChildWhichIsA("BasePart") or obj
    if not adornee then return end
    billboard.Adornee = adornee
    billboard.Parent = obj

    local text = Instance.new("TextLabel")
    text.Size = UDim2.new(1, 0, 1, 0)
    text.BackgroundTransparency = 1
    text.Text = obj.Name
    text.TextColor3 = color
    text.TextStrokeTransparency = 0
    text.TextScaled = false
    text.TextSize = 14 -- можешь менять: 10–20
    text.Font = Enum.Font.GothamBold
    text.Parent = billboard
end

local foundSomething = false

-- 🔍 СКАН
for _, plot in ipairs(plotsFolder:GetChildren()) do
    
    if plot == closestPlot then continue end

    for category, list in pairs(Objects) do
        if Settings[category] then
            
            for _, name in ipairs(list) do
                local obj = plot:FindFirstChild(name, true)
                
                if obj then
                    foundSomething = true
                    print("🔥 НАЙДЕН:", name, "|", category)

                    addESP(obj, Colors[category])
                end
            end

        end
    end
end
